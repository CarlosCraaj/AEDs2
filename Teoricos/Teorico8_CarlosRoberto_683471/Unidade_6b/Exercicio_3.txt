public int maiorElemento(){
    Celula i = topo;
    int aux = topo.elemento;
    i = i.prox;
    int maior = maiorElemento(i, aux);
    return maior;
}

public int maiorElemento(Celula i, int aux){
    if(i != null){
        if(i.elemento > aux){
	    aux = maiorElemento(i.prox, i.elemento);
	}
	
    }
    
    return aux;
}
